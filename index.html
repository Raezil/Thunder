Quick Start

# Install Thunder CLI
git clone https://github.com/Raezil/Thunder.git
cd Thunder
chmod +x install.sh
./install.sh

# Create a new project
thunder init myapp
cd myapp
go mod tidy
# Define your service (example.proto)
syntax = "proto3";
package example;

import "google/api/annotations.proto";
import "graphql.proto";

service Example {
  rpc SayHello(HelloRequest) returns (HelloResponse) {
    option (google.api.http).get = "/v1/example/sayhello";
    option (graphql.schema) = { type: QUERY, name: "sayhello" };
  }
}

message HelloRequest {
  string name = 1 [(graphql.field) = { required: true }];
}

message HelloResponse {
  string message = 1;
}

# Generate REST and GraphQL endpoints
thunder generate --proto=example.proto --enable-graphql
go run ./cmd/app/server/main.go

# Server accessible at:
# gRPC: localhost:50051
# REST: localhost:8080
# GraphQL: localhost:8080/graphql
